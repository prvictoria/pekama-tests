forJson ${JsonPath}
storeEval emailInvalid.length; length
echo ${length}
store 0 MyVar
while storedVars.MyVar < storedVars.length
storeEval emailInvalid[storedVars.MyVar] carname
type name=user_info-email ${carname}
echo ${carname}
type //input[@id='signup-firstname'] firstname
type //input[@id='signup-lastname'] lastname
waitForValue //input[@id='agree_to_tou'] off
click id=agree_to_tou
verifyValue id=agree_to_tou on
verifyValue id=agree_to_tou on
type //input[@id='inputCompany'] Valid Company
type //input[@id='inputPassword'] 0
clickAndWait id=next-button Next Step
waitForText css=li Enter a valid email address.
assertText css=li Enter a valid email address.
store javascript{storedVars.MyVar++;}
echo ${MyVar}
endWhile
store 0 length
echo ${length}
storeEval invalidPassword.length; length
store 0 MyVar
while storedVars.MyVar < storedVars.length
storeEval invalidPassword[storedVars.MyVar] carname
type name=user_info-email 11111111@gmail.com
echo ${carname}
type //input[@id='signup-firstname'] firstname
type //input[@id='signup-lastname'] lastname
waitForValue //input[@id='agree_to_tou'] off
click id=agree_to_tou
verifyValue id=agree_to_tou on
verifyValue id=agree_to_tou on
type //input[@id='inputCompany'] Valid Company
type //input[@id='inputPassword'] ${carname}
clickAndWait id=next-button Next Step
waitForText //form[@id='signup-form']/div/div[3]/div/div[3]/div[2]/div/ul/li This password is too short. It must contain at least 8 characters.
assertText //form[@id='signup-form']/div/div[3]/div/div[3]/div[2]/div/ul/li This password is too short. It must contain at least 8 characters.

echo ${MyVar}
